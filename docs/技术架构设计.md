## 2. 核心组件详细设计

### 2.1 意图识别系统

```python
class IntentRecognitionSystem:
    def __init__(self):
        self.rule_matcher = RuleBasedMatcher()  # 规则匹配器
        self.classifier = LightweightClassifier()  # 轻量级分类器
        self.llm_router = LLMRouter()  # LLM复杂意图路由
    
    def recognize(self, user_input: str) -> Intent:
        # 1. 规则匹配优先
        intent = self.rule_matcher.match(user_input)
        if intent:
            return intent
        
        # 2. 轻量级分类器
        intent = self.classifier.predict(user_input)
        if intent.confidence > 0.8:
            return intent
        
        # 3. LLM复杂意图识别
        return self.llm_router.route(user_input)
```

### 2.3 记忆管理系统

```python
class MemoryManager:
    def __init__(self, redis_client):
        self.redis = redis_client
        self.window_size = 5  # 滑动窗口大小
        self.summarizer = ConversationSummarizer()
    
    async def save_conversation(self, session_id: str, messages: List[Message]):
        # 保存完整对话历史
        await self.redis.rpush(f"session:{session_id}:full", *messages)
        
        # 应用滑动窗口
        if len(messages) > self.window_size:
            recent_messages = messages[-self.window_size:]
            await self.redis.rpush(f"session:{session_id}:recent", *recent_messages)
            
            # 对旧消息进行摘要压缩
            old_messages = messages[:-self.window_size]
            summary = await self.summarizer.summarize(old_messages)
            await self.redis.set(f"session:{session_id}:summary", summary)
    
    async def get_context(self, session_id: str) -> str:
        # 获取最近消息
        recent = await self.redis.lrange(f"session:{session_id}:recent", 0, -1)
        
        # 获取摘要
        summary = await self.redis.get(f"session:{session_id}:summary")
        
        # 组合上下文
        context = ""
        if summary:
            context += f"历史对话摘要：{summary}\n\n"
        if recent:
            context += "最近对话：\n" + "\n".join(recent)
        
        return context
```

# backend/app/api/endpoints.py

# 对话API
@app.post("/api/v1/chat")
async def chat_endpoint(request: ChatRequest):
    """处理用户对话请求"""
    # 1. 获取会话记忆
    memory = await memory_manager.get_context(request.session_id)
    
    # 2. 意图识别
    intent = intent_system.recognize(request.message)
    
    # 3. 路由到对应Agent
    if intent.type == "ordering":
        agent = ordering_agent
    elif intent.type == "after_sales":
        agent = after_sales_agent
    else:
        agent = general_agent
    
    # 4. 执行Agent逻辑
    response = await agent.process(request.message, memory, intent)
    
    # 5. 更新记忆
    await memory_manager.save_conversation(
        request.session_id, 
        request.message, 
        response.message
    )
    
    # 6. 记录监控指标
    await metrics_collector.record_interaction(
        session_id=request.session_id,
        intent=intent.type,
        tool_calls=response.tool_calls,
        success=response.success
    )
    
    return response

# 监控数据API
@app.get("/api/v1/metrics")
async def get_metrics():
    """获取监控指标"""
    return await metrics_collector.get_daily_metrics()

# Human-in-the-Loop API
@app.post("/api/v1/human-intervention")
async def request_human_intervention(request: InterventionRequest):
    """请求人工干预"""
    # 低置信度时触发人工干预
    if request.confidence < 0.7:
        await human_intervention_manager.notify_human(request)
        return {"status": "human_notified", "wait_time": "30秒内响应"}
    
    return {"status": "auto_handled"}

# backend/app/websocket/chat.py

@app.websocket("/ws/chat/{session_id}")
async def websocket_chat(websocket: WebSocket, session_id: str):
    await websocket.accept()
    
    try:
        while True:
            # 接收用户消息
            data = await websocket.receive_json()
            message = data["message"]
            
            # 流式处理
            async for chunk in process_message_streaming(message, session_id):
                await websocket.send_json({
                    "type": "chunk",
                    "content": chunk
                })
            
            await websocket.send_json({"type": "complete"})
            
    except WebSocketDisconnect:
        print(f"客户端断开连接: {session_id}")

## 4. 数据库设计

### 4.1 PostgreSQL表结构

```sql
-- 会话表
CREATE TABLE sessions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 对话历史表
CREATE TABLE conversations (
    id SERIAL PRIMARY KEY,
    session_id UUID REFERENCES sessions(id),
    role VARCHAR(20) NOT NULL, -- 'user' or 'assistant'
    content TEXT NOT NULL,
    tool_calls JSONB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 意图记录表
CREATE TABLE intents (
    id SERIAL PRIMARY KEY,
    session_id UUID REFERENCES sessions(id),
    user_input TEXT NOT NULL,
    detected_intent VARCHAR(50) NOT NULL,
    confidence FLOAT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 工具调用记录表
CREATE TABLE tool_calls (
    id SERIAL PRIMARY KEY,
    session_id UUID REFERENCES sessions(id),
    tool_name VARCHAR(100) NOT NULL,
    parameters JSONB NOT NULL,
    result JSONB,
    success BOOLEAN NOT NULL,
    error_message TEXT,
    duration_ms INTEGER,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 监控指标表
CREATE TABLE metrics (
    id SERIAL PRIMARY KEY,
    date DATE NOT NULL,
    tool_call_count INTEGER DEFAULT 0,
    user_correction_count INTEGER DEFAULT 0,
    retrieval_failure_count INTEGER DEFAULT 0,
    avg_response_time_ms INTEGER,
    success_rate FLOAT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(date)
);
```
